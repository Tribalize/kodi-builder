# Kodi 22 Piers Builder Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make Kodi 22 Piers, sourced from Kodi's rolling `master` branch, the default Android APK target while retaining the Omega, Nexus, and Matrix choices.

**Architecture:** Add one workflow profile-resolution step that separates the user-facing Kodi release from the upstream Git ref and Android toolchain. Keep all rolling-version knowledge in that profile, fingerprint upstream `tools/depends` for safe cache reuse, and enforce the contract with a dependency-free repository validator.

**Tech Stack:** GitHub Actions YAML, Bash, Python 3 standard library, PowerShell, Kodi unified depends, Android SDK/NDK

---

## File Map

- Create `scripts/validate-kodi22-builder.py`: cross-platform static contract checks for the workflow and user-facing defaults.
- Modify `.github/workflows/build-kodi.yml`: Piers input/profile resolution, rolling source metadata, Android 37.0 tools, profile-aware signing, caching, diagnostics, and prereleases.

- Modify `kodi-config/addons.txt`: make the Piers addon channel the primary example while retaining legacy channel guidance.
- Modify `kodi-config/sources.xml`: change the official repository example from Omega to Piers.

Create `scripts/validate-kodi22-builder.py` with:

```python
#!/usr/bin/env python3
"""Validate the Kodi 22 Piers builder contract without running Android builds."""

from __future__ import annotations

import argparse
import re
from pathlib import Path


ROOT = Path(__file__).resolve().parents[1]


def read(path: str) -> str:
    return (ROOT / path).read_text(encoding="utf-8")


def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument(
        "--scope", choices=("all", "workflow", "docs"), default="all"
    )
    scope = parser.parse_args().scope

    workflow = read(".github/workflows/build-kodi.yml")
    readme = read("README.md")
    addons = read("kodi-config/addons.txt")
    sources = read("kodi-config/sources.xml")
    preflight = read("scripts/local-gha-preflight.ps1")

    checks: list[tuple[str, str, bool]] = []

    def check(group: str, label: str, condition: bool) -> None:
        checks.append((group, label, condition))

    check("workflow", "Piers is the workflow default", "default: 'Piers'" in workflow)
    for release in ("Piers", "Omega", "Nexus", "Matrix"):
        check(
            "workflow",
            f"{release} remains selectable",
            f"          - {release}\n" in workflow,
        )
    check(
        "workflow",
        "raw master is not exposed as a release choice",
        "          - master\n" not in workflow,
    )
    check(
        "workflow",
        "ABI default is an exact selectable value",
        "        default: 'arm64-v8a'" in workflow
        and "          - arm64-v8a\n" in workflow,
    )
    check(
        "workflow",
        "Piers resolves to rolling master and Android 37.0",
        bool(
            re.search(
                r"Piers\).*?SOURCE_REF=master.*?ADDON_CHANNEL=piers"
                r".*?ANDROID_PLATFORM=37\.0.*?ANDROID_BUILD_TOOLS=37\.0\.0"
                r".*?IS_PRERELEASE=true",
                workflow,
                re.DOTALL,
            )
        ),
    )
    check(
        "workflow",
        "legacy Android 34 and Piers Android 37.0 SDKs are installed",
        "platforms;android-34 build-tools;34.0.0" in workflow
        and "platforms;android-37.0 build-tools;37.0.0" in workflow,
    )
    check("workflow", "NDK r28c remains configured", "android-ndk-r28c" in workflow)
    check("workflow", "Java 17 remains configured", "java-version: '17'" in workflow)
    check(
        "workflow",
        "source checkout uses the resolved ref",
        workflow.count("steps.kodi.outputs.source_ref") >= 4,
    )
    check(
        "workflow",
        "depends cache uses the upstream depends fingerprint",
        "steps.source.outputs.depends_fingerprint" in workflow,
    )
    check(
        "workflow",
        "signing uses the selected build-tools directory",
        'BUILD_TOOLS="$ANDROID_HOME/build-tools/${{ steps.kodi.outputs.android_build_tools }}"'
        in workflow
        and "$ANDROID_HOME/build-tools/34.0.0/zipalign" not in workflow,
    )
    check(
        "workflow",
        "Piers releases are marked prerelease",
        "prerelease: ${{ steps.kodi.outputs.is_prerelease == 'true' }}" in workflow,
    )
    check(
        "workflow",
        "workflow runs the contract validator",
        "python3 scripts/validate-kodi22-builder.py" in workflow,
    )

    check(
        "docs",
        "README identifies Piers as Kodi 22 prerelease",
        "Piers" in readme and "Kodi 22" in readme and "prerelease" in readme.lower(),
    )
    check(
        "docs",
        "addon examples use the Piers channel",
        "https://mirrors.kodi.tv/addons/piers/" in addons,
    )
    check(
        "docs",
        "source example uses the Piers channel",
        "<name>Kodi Piers Repo</name>" in sources
        and "https://mirrors.kodi.tv/addons/piers/" in sources,
    )
    check(
        "docs",
        "local preflight dispatches Piers",
        preflight.count("--input branch=Piers") == 2
        and "--input branch=Omega" not in preflight,
    )

    selected = [item for item in checks if scope == "all" or item[0] == scope]
    failures = [label for _, label, passed in selected if not passed]
    for _, label, passed in selected:
        print(f"[{'PASS' if passed else 'FAIL'}] {label}")
    if failures:
        print(f"\n{len(failures)} validation check(s) failed.")
        return 1
    print(f"\nAll {len(selected)} validation checks passed.")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

```yaml
      - name: Resolve Kodi build profile
        id: kodi
        shell: bash
        run: |
          set -euo pipefail
          RELEASE="${{ github.event.inputs.branch }}"
          case "$RELEASE" in
            Piers)
              SOURCE_REF=master
              ADDON_CHANNEL=piers
              ANDROID_PLATFORM=37.0
              ANDROID_BUILD_TOOLS=37.0.0
              IS_PRERELEASE=true
              ;;
            Omega)
              SOURCE_REF=Omega
              ADDON_CHANNEL=omega
              ANDROID_PLATFORM=34
              ANDROID_BUILD_TOOLS=34.0.0
              IS_PRERELEASE=false
              ;;
            Nexus)
              SOURCE_REF=Nexus
              ADDON_CHANNEL=nexus
              ANDROID_PLATFORM=34
              ANDROID_BUILD_TOOLS=34.0.0
              IS_PRERELEASE=false
              ;;
            Matrix)
              SOURCE_REF=Matrix
              ADDON_CHANNEL=matrix
              ANDROID_PLATFORM=34
              ANDROID_BUILD_TOOLS=34.0.0
              IS_PRERELEASE=false
              ;;
            *)
              echo "Unsupported Kodi release: $RELEASE" >&2
              exit 1
              ;;
          esac
          {
            echo "release=$RELEASE"
            echo "source_ref=$SOURCE_REF"
            echo "addon_channel=$ADDON_CHANNEL"
            echo "android_platform=$ANDROID_PLATFORM"
            echo "android_build_tools=$ANDROID_BUILD_TOOLS"
            echo "is_prerelease=$IS_PRERELEASE"
          } >> "$GITHUB_OUTPUT"
          echo "Kodi $RELEASE -> xbmc/xbmc $SOURCE_REF"
          echo "Android platform $ANDROID_PLATFORM / build-tools $ANDROID_BUILD_TOOLS"
```

```bash
          SOURCE_REF="${{ steps.kodi.outputs.source_ref }}"
          if [ -d ~/kodi/.git ]; then
            echo "Updating existing source..."
            git -C ~/kodi fetch origin "$SOURCE_REF"
            git -C ~/kodi checkout "$SOURCE_REF"
            git -C ~/kodi reset --hard "origin/$SOURCE_REF"
            git -C ~/kodi clean -fd
          else
            echo "Cloning Kodi $SOURCE_REF..."
            git clone --depth=1 -b "$SOURCE_REF" https://github.com/xbmc/xbmc ~/kodi
          fi
          SOURCE_SHA=$(git -C ~/kodi rev-parse HEAD)
          DEPENDS_FINGERPRINT=$(git -C ~/kodi ls-tree -r HEAD tools/depends | sha256sum | cut -d' ' -f1)
          echo "source_sha=$SOURCE_SHA" >> "$GITHUB_OUTPUT"
          echo "depends_fingerprint=$DEPENDS_FINGERPRINT" >> "$GITHUB_OUTPUT"
          echo "Source ready (${SOURCE_SHA:0:12})"
          echo "Depends fingerprint: $DEPENDS_FINGERPRINT"
```
